import Bull, { Job, Queue, JobOptions } from 'bull';
import { log } from '../middleware/logger';

// 1. Configuración de conexión única
// Si estamos en Railway, process.env.REDIS_URL estará definida.
// Bull y ioredis entienden la URL completa (redis://user:pass@host:port) nativamente.
const redisConfig = process.env.REDIS_URL 
  ? process.env.REDIS_URL 
  : { host: '127.0.0.1', port: 6379 };

const defaultOptions: JobOptions = {
  attempts: 3,
  backoff: { type: 'exponential', delay: 5000 },
};

// 2. Instancias de Colas
export const qValidar   = new Bull('rodaid:cit:validar',   { redis: redisConfig, defaultJobOptions: defaultOptions });
export const qFinalizar = new Bull('rodaid:cit:finalizar', { redis: redisConfig, defaultJobOptions: defaultOptions });
export const qNotif     = new Bull('rodaid:notif',         { redis: redisConfig, defaultJobOptions: defaultOptions });
export const qExpire    = new Bull('rodaid:cit:expirar',   { redis: redisConfig, defaultJobOptions: defaultOptions });

// 4. API de encolado simplificada
export async function encolarValidacion(citId: string, venceEn: Date): Promise<string | undefined> {
  const delayMs = Math.max(0, venceEn.getTime() - Date.now());
  const job = await qValidar.add({ citId }, { delay: delayMs, jobId: `validar:${citId}` });
  return String(job.id);
}

export async function encolarFinalizar(citId: string, propietarioWallet?: string): Promise<string | undefined> {
  const job = await qFinalizar.add({ citId, propietarioWallet }, { jobId: `finalizar:${citId}` });
  return String(job.id);
}
